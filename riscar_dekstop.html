Dynamo Integration
Python script · OneDrive Excel ↔ Revit color mapping
# Dynamo – RPN Color Override via OneDrive
import clr
clr.AddReference('RevitAPI')
from Autodesk.Revit.DB import *
rpn_data = read_excel("OneDrive/RFMEA_Djoglo.xlsx")
for element in elements:
  rpn = rpn_data[element.Id]
  if rpn > 200: color = RED
  elif rpn > 80: color = YELLOW
  else: color = GREEN
  override_color(element, color)
